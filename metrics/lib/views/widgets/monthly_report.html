  getTime(Timestamp time) {
    var date = DateTime.fromMillisecondsSinceEpoch(time.millisecondsSinceEpoch);
    String parsedObject = date.day.toString() +
        "/" +
        date.month.toString() +
        "/" +
        date.year.toString();
    return parsedObject;
  }
